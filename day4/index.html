Immediately Invoked Function Expression

  • IIFE is an anonymous function that is created and then immediately invoked
  • Design pattern which produces a lexical scope using JavaScript's function scoping
  • Used for avoiding variable hoisting and protects against polluting the global namespace

Constructor Pattern

function Property (name, city, state) {

  this.name = name;
  this.city = city;
  this.state = state;

  this.displayPropertyInfo = function () {
    return this.name + ', ' + this.city +  ', ' + this.state;
  };
}

var prop = new Property('Pensacola Place', 'Chicago', 'IL');

console.log(prop.name);
console.log(prop.displayPropertyInfo());

Module Pattern

Using Object Literal Notation

var propertyModule = {

  name: 'Pensacola Place',

  details: {
    city: 'Chicago',
    state: 'IL';
  },

  displayPropertyInfo: function () {
    return this.name + ', ' + this.details.city;
  }
};

console.log(propertyModule.name);
console.log(propertyModule.displayPropertyInfo());

Using IIFE (Immediately Invoked Function Expression)

var propertyModule = (function () {

  /* private members */
  var city = 'Chicago',
      state = 'IL';

  var displayDetails = function () {
    console.log(city, state);
  };

  /* public members */
  return {
    name: 'Pensacola Place',
    displayPropertyInfo: function () {
        return name + ', ' + city;
    }
  };
})();

                    

/* public members */

console.log(propertyModule.name);

console.log(propertyModule.displayPropertyInfo());

propertyModule.name = 'Orange Apt';
